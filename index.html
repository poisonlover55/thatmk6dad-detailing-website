        // JavaScript for handling form submission (client-side only, no actual backend)
        document.querySelector('form').addEventListener('submit', function(event) {
            event.preventDefault(); // Prevent default form submission

            const messageBox = document.getElementById('messageBox');
            messageBox.classList.remove('hidden'); // Show the message box
            messageBox.classList.remove('bg-red-100', 'text-red-800');
            messageBox.classList.add('bg-green-100', 'text-green-800');
            messageBox.textContent = 'Your request has been sent successfully! We will contact you soon.';

            // Optionally, clear the form after submission
            this.reset();

            // Hide the message after a few seconds
            setTimeout(() => {
                messageBox.classList.add('hidden');
            }, 5000);
        });

        // Function to call Gemini API
        async function callGeminiAPI(prompt, outputElementId, loadingElementId) {
            const outputElement = document.getElementById(outputElementId);
            const loadingElement = document.getElementById(loadingElementId);

            outputElement.classList.add('hidden'); // Hide previous output
            loadingElement.classList.remove('hidden'); // Show loading indicator
            outputElement.textContent = ''; // Clear previous text

            let chatHistory = [];
            chatHistory.push({ role: "user", parts: [{ text: prompt }] });

            const payload = { contents: chatHistory };
            const apiKey = ""; // Leave as empty string, Canvas will provide it
            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

            try {
                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                const result = await response.json();

                if (result.candidates && result.candidates.length > 0 &&
                    result.candidates[0].content && result.candidates[0].content.parts &&
                    result.candidates[0].content.parts.length > 0) {
                    const text = result.candidates[0].content.parts[0].text;
                    outputElement.textContent = text;
                    outputElement.classList.remove('hidden'); // Show the generated text
                } else {
                    outputElement.textContent = 'Sorry, I could not generate a response. Please try again.';
                    outputElement.classList.remove('hidden');
                }
            } catch (error) {
                console.error('Error calling Gemini API:', error);
                outputElement.textContent = 'An error occurred while fetching data. Please try again later.';
                outputElement.classList.remove('hidden');
            } finally {
                loadingElement.classList.add('hidden'); // Hide loading indicator
            }
        }

        // Event listener for "Get a Car Care Tip" button
        document.getElementById('generateTipBtn').addEventListener('click', function() {
            const prompt = "Provide a concise and helpful car care tip (1-2 sentences) for maintaining a vehicle's appearance. Do not include any introductory or concluding phrases, just the tip itself.";
            callGeminiAPI(prompt, 'tipText', 'tipLoading');
            document.getElementById('tipOutput').classList.remove('hidden'); // Ensure the output container is visible
        });

        // Event listener for "Learn More About Ceramic Coatings" button
        document.getElementById('ceramicInfoBtn').addEventListener('click', function() {
            const prompt = "Explain what ceramic coatings are for cars and their main benefits in a simple, easy-to-understand paragraph (3-4 sentences). Focus on protection, shine, and maintenance. Do not include any introductory or concluding phrases.";
            callGeminiAPI(prompt, 'ceramicInfoText', 'ceramicInfoLoading');
            document.getElementById('ceramicInfoOutput').classList.remove('hidden'); // Ensure the output container is visible
        });
    </script>
